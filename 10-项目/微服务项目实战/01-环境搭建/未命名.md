配置nacos
第一步创建工作目录
 # 2. 创建 docker-compose.yaml 文件（内容如下）：

  version: "2"
  services:
    nacos:
      image: crpi-5yw4kk2bxbk3nj6k.cn-hangzhou.personal.cr.aliyuncs.com/01xq/nacos-server:v2.1.0
      container_name: nacos-standalone
      environment:
      - PREFER_HOST_MODE=hostname
      - MODE=standalone
      - JVM_XMS=256m
      - JVM_XMX=256m
      - JVM_XMN=128m
      - JVM_MS=32m
      - JVM_MMS=80m
      volumes:
      - ./standalone-logs/:/home/nacos/logs
      - ./init.d/custom.properties:/home/nacos/init.d/custom.properties
      ports:
      - "8848:8848"
      - "9848:9848"



  # 3. 创建 custom.properties 文件（空文件即可，用于挂载）
  # Windows上可以用以下命令创建空文件：
  type nul > D:/DockerServices/nacos/init.d/custom.properties

  # 4. 启动容器
  cd D:/DockerServices/nacos
  docker-compose up -d
