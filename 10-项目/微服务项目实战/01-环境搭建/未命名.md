  # 1. 创建工作目录

  mkdir -p D:/DockerServices/redis/conf

  # 2. 下载 redis.conf 配置文件到 conf 目录
  # 或者直接创建基础配置文件（根据文档内容，关键配置如下）：

  # 基础配置（保存为 conf/redis.conf）：
  bind 0.0.0.0
  protected-mode no
  port 6379
  timeout 0
  tcp-keepalive 300
  daemonize no
  pidfile /var/run/redis_6379.pid
  loglevel notice
  logfile ""
  databases 16
  save 900 1
  save 300 10
  save 60 10000
  stop-writes-on-bgsave-error yes
  rdbcompression yes
  rdbchecksum yes
  dbfilename dump.rdb
  dir /data
  appendonly no

  # 3. 启动 Redis 容器
  cd D:/DockerServices/redis
  docker run -id --name redis-6379 \
      -p 6379:6379 \
      -v $PWD/data:/data \
      -v $PWD/conf/redis.conf:/etc/redis/redis.conf \
      crpi-5yw4kk2bxbk3nj6k.cn-hangzhou.personal.cr.aliyuncs.com/01xq/redis:6.2.7 \
      redis-server /etc/redis/redis.conf
