    student {
        BIGINT id PK "自增主键"
        VARCHAR student_no UK "学号（唯一）"
        VARCHAR real_name "真实姓名"
        DATETIME created_at "创建时间"
    }

    login_log {
        BIGINT id PK "自增主键"
        VARCHAR student_no "学号"
        VARCHAR real_name "姓名"
        TINYINT is_success "登录是否成功（0/1）"
        DATETIME login_at "登录时间"
    }

    note {
        BIGINT id PK "自增主键"
        BIGINT student_id FK "关联 student.id"
        TEXT content "笔记内容"
        TINYINT is_deleted "逻辑删除标志（0/1）"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
    }

    tag {
        BIGINT id PK "自增主键"
        BIGINT student_id FK "关联 student.id"
        VARCHAR name "标签名称"
        DATETIME created_at "创建时间"
    }

    note_tag {
        BIGINT id PK "自增主键"
        BIGINT note_id FK "关联 note.id"
        BIGINT tag_id FK "关联 tag.id"
    }

    student ||--o{ note : "拥有"
    student ||--o{ tag : "创建"
    student ||--o{ login_log : "产生"
    note ||--o{ note_tag : "包含"
    tag ||--o{ note_tag : "关联"
